Java基于queue的多线程提取SVN信息泄露程序

发表于 2015-12-16   |   分类于 编程开发   |  

0x01 背景

引上一篇博文，对于多线程程序来说，不管任何编程语言，生产者和消费者模型都是最经典的。就像学习每一门编程语言你首先要输出Hello World一样，都是经典的例子。这里使用queue队列作为仓储，将生产者和消费者联系起来并模拟实现一个多线程提取SVN信息泄露漏洞的程序。

0x02 基于Queue的多线程提取SVN信息泄露的实现

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLConnection;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.LinkedBlockingQueue;
import ReadFiles.ReadFromFile;
import WriteFiles.WriteToFile;

/**
 * 
 * ClassName: svnExtract 
 * @Description 获取域名list中存在svn信息泄露的域名
 * @author HackBraid
 * @date 2015年11月4日
 */
public class svnExtract {

public static BlockingQueue<String> svnQueue = new LinkedBlockingQueue<String>();
public static void main(String[] args) {
    // TODO Auto-generated method stub
    svnExtract svnExtract=new svnExtract();
    long startTime=System.currentTimeMillis(); 
    svnExtract.svnProducer(svnQueue, "C:/Users/Administrator/Desktop/domain.txt");
    long nextTime=System.currentTimeMillis(); 
    System.out.println("生成svnQueue队列的运行时间： "+(nextTime-startTime)+"ms");
    ThreadGroup tg=new ThreadGroup("svnExtract");
    svnConsumer svnConsumer=svnExtract.new svnConsumer();
    for (int i = 0; i < 100; i++) {
        Thread thread=new Thread(tg,svnConsumer);
        thread.start();
    }
    while (true) {
        if(tg.activeCount()==0){
            break;
        }
    }
    long endTime=System.currentTimeMillis(); 
    System.out.println("程序总运行时间： "+(endTime-startTime)+"ms");
}

/**
 * 
 * @Description 根据文件路径将文件里的url填充到svnQueue中
 * @param svnQueue
 * @param filePath
 * @return svnQueue 
 * @author HackBraid
 * @date 2015年11月4日
 */
private BlockingQueue<String> svnProducer(BlockingQueue<String> svnQueue, String filePath) {
    String urlTmp;
    ReadFromFile readFromFile = new ReadFromFile();
    List urlList = new ArrayList<String>();
    urlList = readFromFile.readFileByLines(filePath);
    for (int i = 0; i < urlList.size(); i++) {
        urlTmp="http://"+urlList.get(i)+"/.svn/entries";
        svnQueue.offer(urlTmp);
    }
    return svnQueue;

}

/**
 * 
 * ClassName: svnConsumer 
 * @Description 处理svnQueue中的url，判断是否存在svn信息泄露
 * @author HackBraid
 * @date 2015年11月4日
 */
public class svnConsumer implements Runnable{

    @Override
    public void run() {
        while (true) {
            String url=svnQueue.poll();
            if(url == null){
                return;
            }
            try{
            URL u = new URL(url);
            URLConnection connection = u.openConnection();
            HttpURLConnection httpURLConnection = (HttpURLConnection) connection;
            httpURLConnection.setRequestProperty("Accept-Charset", "utf-8");
            httpURLConnection.setRequestProperty("Accept-Encoding", "identity");
            httpURLConnection.setRequestProperty("Content-Type",
                    "application/x-www-form-urlencoded");
            httpURLConnection.setReadTimeout(2000);
            httpURLConnection.setConnectTimeout(2000);
            if (((httpURLConnection.getResponseCode() == 200)||(httpURLConnection.getResponseCode() == 403)||(httpURLConnection.getResponseCode() == 404))) {
                //获取返回内容    
                InputStream in = null;
                InputStreamReader inr = null;
                BufferedReader reader = null;
                StringBuffer resultBuffer = new StringBuffer();
                String tempLine = null;
                 try{
                 in = httpURLConnection.getInputStream();
                 //先对数据流进行utf-8转码，解决乱码
                 inr = new InputStreamReader(in,"utf-8");
                 reader = new BufferedReader(inr);
                 while((tempLine = reader.readLine())!=null){
                     if(tempLine.indexOf("ir")>0&&tempLine.length()<5){
                         WriteToFile.fWriter("C:/Users/Administrator/Desktop/svnSuccess.txt", url);
                         break;
                     }
                 }
                 }finally{
                 if(reader!=null){
                 reader.close();
                 }
                 if(inr!=null){
                 inr.close();
                 }
                 if(in!=null){
                 in.close();
                 }
                 }
            }
            else{
            }
        } catch (IOException e) {
        }

        }
    }

}
}
